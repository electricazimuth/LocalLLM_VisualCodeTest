```html
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Aquarium Simulation</title>
    <style>
        body {
            margin: 0;
            overflow: hidden;
            background-color: #e0f7fa; /* Light blue background */
        }

        canvas {
            display: block;
            margin: 20px auto;
            border: 1px solid #888;
            background-color: rgb(207, 247, 255); /* blue water */
        }
    </style>
</head>
<body id="body">

    <canvas id="simulationCanvas" width="800" height="600"></canvas>

    <script>
        const canvas = document.getElementById('simulationCanvas');
        const ctx = canvas.getContext('2d');
        const W = canvas.width;
        const H = canvas.height;

        // Fish class
        class Fish {
            constructor(x, y, size, color, speed, angle) {
                this.x = x;
                this.y = y;
                this.size = size;
                this.color = color;
                this.speed = speed;
                this.angle = angle;
                this.vx = Math.cos(this.angle) * this.speed;
                this.vy = Math.sin(this.angle) * this.speed;
            }

            update(deltaTime) {
                this.x += this.vx * deltaTime;
                this.y += this.vy * deltaTime;

                // Boundary collision
                if (this.x < 0 || this.x > W - this.size) {
                    this.vx *= -1;
                    this.angle += Math.PI / 4; // Small angle adjustment
                }
                if (this.y < 0 || this.y > H - this.size) {
                    this.vy *= -1;
                    this.angle += Math.PI / 4;
                }

                // Wandering behavior (small random angle change)
                this.angle += Math.random() * 0.02 - 0.01;
            }

            draw() {
                ctx.beginPath();
                ctx.arc(this.x, this.y, this.size, 0, Math.PI * 2);
                ctx.fillStyle = this.color;
                ctx.fill();
                ctx.closePath();
            }
        }

        // Fish initialization
        const numFish = 15;
        const fish = [];
        for (let i = 0; i < numFish; i++) {
            const x = Math.random() * (W - 50);
            const y = Math.random() * (H - 50);
            const size = Math.random() * 10 + 5;
            const color = `rgb(${Math.floor(Math.random() * 255), Math.floor(Math.random() * 255), Math.floor(Math.random() * 255)})`;
            const speed = Math.random() * 5 + 2;
            const angle = Math.random() * Math.PI * 2; // Random initial angle
            fish.push(new Fish(x, y, size, color, speed, angle));
        }

        // Animation loop
        function gameLoop(timestamp) {
            const deltaTime = timestamp - lastTimestamp;

            ctx.clearRect(0, 0, W, H);

            fish.forEach(fish => {
                fish.update(deltaTime);
                fish.draw();
            });

            lastTimestamp = timestamp;
            requestAnimationFrame(gameLoop);
        }

        // Start the loop
        let lastTimestamp = 0;
        gameLoop(0);

    </script>
</body>
</html>
```

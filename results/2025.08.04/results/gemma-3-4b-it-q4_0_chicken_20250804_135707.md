    <script>
        // --- YOUR JAVASCRIPT CODE GOES HERE ---

        // Canvas setup
        const canvas = document.getElementById('simulationCanvas');
        const ctx = canvas.getContext('2d');
        canvas.width = 512;
        canvas.height = 512;

        // Chicken object
        class Chicken {
            constructor(x, y) {
                this.x = x;
                this.y = y;
                this.radius = 20;
                this.direction = Math.random() * 2 * Math.PI; // Initial direction
                this.velocity = 3;
                this.color = 'yellow';
                this.laying = false;
                this.layingTimer = 0;
                this.layingTimeout = 50; // milliseconds
            }

            update() {
                if (this.laying) {
                    this.layingTimer++;
                    if (this.layingTimer > this.layingTimeout) {
                        this.laying = false;
                        this.layingTimer = 0;
                    }
                    return;
                }

                // Update position based on direction and velocity
                this.x += this.velocity * Math.cos(this.direction);
                this.y += this.velocity * Math.sin(this.direction);

                // Boundary handling
                if (this.x < this.radius) {
                    this.x = this.radius;
                    this.direction = (Math.random() - 0.5) * Math.PI; // Turn left or right
                }
                if (this.x > canvas.width - this.radius) {
                    this.x = canvas.width - this.radius;
                    this.direction = (Math.random() - 0.5) * Math.PI; // Turn left or right
                }
                if (this.y < this.radius) {
                    this.y = this.radius;
                    this.direction = (Math.random() - 0.5) * Math.PI; // Turn left or right
                }
                if (this.y > canvas.height - this.radius) {
                    this.y = canvas.height - this.radius;
                    this.direction = (Math.random() - 0.5) * Math.PI; // Turn left or right
                }

                // Egg laying logic
                if (Math.random() < 0.02) { // Low probability of laying
                    this.laying = true;
                    this.layingTimer = 0;
                }
            }

            draw() {
                ctx.beginPath();
                ctx.arc(this.x, this.y, this.radius, 0, 2 * Math.PI);
                ctx.fillStyle = this.color;
                ctx.fill();
                ctx.closePath();

                if (this.laying) {
                    ctx.fillStyle = 'gold';
                    ctx.beginPath();
                    ctx.arc(this.x, this.y, this.radius / 2, 0, 2 * Math.PI);
                    ctx.fill();
                    ctx.closePath();
                }
            }
        }

        // Egg object
        class Egg {
            constructor(x, y) {
                this.x = x;
                this.y = y;
                this.radius = 10;
                this.color = 'white';
            }

            draw() {
                ctx.beginPath();
                ctx.arc(this.x, this.y, this.radius, 0, 2 * Math.PI);
                ctx.fillStyle = this.color;
                ctx.fill();
                ctx.closePath();
            }
        }

        // Simulation setup
        const chicken = new Chicken(canvas.width / 2, canvas.height / 2);
        const eggs = [];

        // Animation loop
        function animate() {
            requestAnimationFrame(animate);

            // Clear canvas
            ctx.clearRect(0, 0, canvas.width, canvas.height);

            // Update chicken
            chicken.update();

            // Draw chicken
            chicken.draw();

            // Draw eggs
            eggs.forEach(egg => egg.draw());

            // Add new egg if laying
            if (chicken.laying) {
                const egg = new Egg(chicken.x, chicken.y);
                eggs.push(egg);
            }
        }

        // Start animation
        animate();

    </script>

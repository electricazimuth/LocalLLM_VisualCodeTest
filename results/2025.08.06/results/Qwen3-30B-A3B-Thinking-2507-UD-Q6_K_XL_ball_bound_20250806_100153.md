                const ball1 = balls[i];
                const cellX = Math.floor(ball1.x / cellWidth);
                const cellY = Math.floor(ball1.y / cellHeight);

                    for (let dy = -1; dy <= 1; dy++) {

                        if (nx >= 0 && nx < GRID_COLS &&
